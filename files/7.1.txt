// simple calculator

import javax.swing.*;
import java.awt.*;
import java.awt.event.*;

class Calculator implements ActionListener{
JFrame Frame;
JLabel Num1, Num2, lResult;
JTextField tNum1,tNum2, Result;
JButton addButton, subButton, mulButton, divButton, equalButton, clearButton;


public Calculator() {
Frame=new JFrame("Calculator");
Frame.setSize(500, 500);


Num1=new JLabel("Enter First Number: ");
Num1.setBounds(30,90,150,50);
Frame.add(Num1);


Num2=new JLabel("Enter Sec Number: ");
Num2.setBounds(30,140,150,50);
Frame.add(Num2);

lResult=new JLabel("Answer: ");
lResult.setBounds(30,190,100,50);
Frame.add(lResult);


tNum1 = new JTextField();
tNum1.setBounds(150, 100, 200, 30); 
Frame.add(tNum1);


tNum2 = new JTextField();
tNum2.setBounds(150, 150, 200, 30); 
Frame.add(tNum2);

Result = new JTextField();
Result.setBounds(150, 200, 200, 30);
Result.setEditable(false); 
Frame.add(Result);



JPanel buttonPanel = new JPanel();
buttonPanel.setBounds(150, 250, 200, 200); 

addButton = new JButton("+");
subButton = new JButton("-");
mulButton = new JButton("*");
divButton = new JButton("/");
clearButton = new JButton("Clear");

buttonPanel.add(addButton);
buttonPanel.add(subButton);
buttonPanel.add(mulButton);
buttonPanel.add(divButton);
buttonPanel.add(clearButton);

addButton.addActionListener(this);
subButton.addActionListener(this);
mulButton.addActionListener(this);
divButton.addActionListener(this);
clearButton.addActionListener(this);

Frame.add(buttonPanel);
Frame.setLayout(null);
Frame.setVisible(true);
}


public void actionPerformed(ActionEvent e)
{
int a=Integer.parseInt(tNum1.getText());
int b=Integer.parseInt(tNum2.getText());
if(e.getSource()==addButton)
{
int c=a+b;
//JOptionPane.showMessageDialog(Frame,c);
Result.setText(c + "");
}

else if(e.getSource()==subButton)
{
int c=a-b;
//JOptionPane.showMessageDialog(Frame,c);
Result.setText(c + "");
}

else if(e.getSource()==mulButton)
{
int c=a*b;
//JOptionPane.showMessageDialog(Frame,c);
Result.setText(c + "");
}

else if(e.getSource()==divButton)
{
int c=a/b;
//JOptionPane.showMessageDialog(Frame,c);
//Result.setText(c);
Result.setText(c + "");
}

else if(e.getSource()==clearButton)
{
tNum1.setText("");
tNum2.setText("");
Result.setText("");
}

}

public static void main(String args[]) {
new Calculator();
}
}
