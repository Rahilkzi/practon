// concept of interface interfaces

interface A
{
public void set();
public void get();
}
class B implements A
{
public void set()
{
System.out.println("This is implemented set");
}
public void get()
{
System.out.println("This is implemented get");
}
public void check()
{
System.out.println("This is normal method of class B");
}
}
class InterfaceDemo1
{
public static void main(String args[])
{
B b1=new B();
System.out.println("calling from main method");
b1.set();
b1.get();
b1.check();
}
}
